## Structure

### Single form elements

```html
<div class="tna-form-item"> <!-- Optional: "tna-form-item--error" -->

  <div class="tna-form-item__contents">

    <h4 class="tna-form-item__heading tna-heading-m">
      <label class="tna-form-item__label" for="ELEMENT-ID">
        <!-- Field title -->
      </label>
    </h4>

    <!-- Optional hint
    <p id="ELEMENT-ID-hint" class="tna-form-item__hint">
      [Hint text]
    </p>
    -->

    <!-- Optional error
    <p id="ELEMENT-ID-error" class="tna-form-item__error">
      <span class="tna-!--visually-hidden">Error:</span> [Error text]
    </p>
    -->

  </div>

  <!-- Field id="ELEMENT-ID"
    If hint:
      aria-describedby="tna-form__ELEMENT-ID-hint"
    If error:
      aria-describedby="tna-form__ELEMENT-ID-error"
    If hint and error:
      aria-describedby="tna-form__ELEMENT-ID-hint tna-form__ELEMENT-ID-error"
  -->

</div>
```

### Fieldset (multiple inputs - e.g. checkboxes and radios)

```html
<div class="tna-form-item"> <!-- Optional: "tna-form-item--error" -->

  <fieldset class="tna-form__fieldset"><!-- Fieldset
      If hint:
        aria-describedby="tna-form__ELEMENT-ID-hint"
      If error:
        aria-describedby="tna-form__ELEMENT-ID-error"
      If hint and error:
        aria-describedby="tna-form__ELEMENT-ID-hint tna-form__ELEMENT-ID-error"
    -->

    <div class="tna-form-item__contents">

      <legend class="tna-form__legend">
        <h4 class="tna-form-item__heading tna-heading-m">
          <!-- Field title -->
        </h4>
      </legend>

      <!-- Optional hint
      <p id="ELEMENT-ID-hint" class="tna-form-item__hint">
        [Hint text]
      </p>
      -->

      <!-- Optional error
      <p id="ELEMENT-ID-error" class="tna-form-item__error">
        <span class="tna-!--visually-hidden">Error:</span> [Error text]
      </p>
      -->

    </div>
  </fieldset>

  <!-- Fields -->
  
</div>
```
